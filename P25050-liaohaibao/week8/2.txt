# 2、给定一个整型列表，请实现从其中找出2个数的和为某一个指定的值？
# 如：lst =[1,5,2,7,4,9]，指定的目标值为11，可以从中找出 2和9之和为11
def numbers(lst:list,k:int):
    l = len(lst)
    for i in range(l):
        for j in range(i,l):
            if lst[i] + lst[j] == k:
                return (i,j)
    return None

lst = [1,5,2,7,4,9]
key = 11
print(numbers(lst,key))
